15// This function initializes the DMA to a known state.



16//



17void DMAInitialize(void)



18{



19	EALLOW;



20



21	// Perform a hard reset on DMA



22	DmaRegs.DMACTRL.bit.HARDRESET = 1;



23



24	// Allow DMA to run free on emulation suspend



25	DmaRegs.DEBUGCTRL.bit.FREE = 1;



26	



27	EDIS;



28}

31void DMACH1AddrConfig(volatile Uint16 *DMA_Dest,volatile Uint16 *DMA_Source)



32{



33	EALLOW;



34	// Set up SOURCE address:



35	DmaRegs.CH1.SRC_BEG_ADDR_SHADOW = (Uint32)DMA_Source;	// Point to beginning of source buffer



36	DmaRegs.CH1.SRC_ADDR_SHADOW =     (Uint32)DMA_Source;



37



38	// Set up DESTINATION address:



39	DmaRegs.CH1.DST_BEG_ADDR_SHADOW = (Uint32)DMA_Dest;	    // Point to beginning of destination buffer



40	DmaRegs.CH1.DST_ADDR_SHADOW =     (Uint32)DMA_Dest;



41



42	



43	EDIS;



44}

71void DMACH1WrapConfig(Uint16 srcwsize, int16 srcwstep, Uint16 deswsize, int16 deswstep)

86void DMACH1ModeConfig(Uint16 persel, Uint16 perinte, Uint16 oneshot, Uint16 cont, Uint16 synce, Uint16 syncsel, Uint16 ovrinte, Uint16 datasize, Uint16 chintmode, Uint16 chinte)



87{



88	EALLOW;



89	    



90	// Set up MODE Register:



91	DmaRegs.CH1.MODE.bit.PERINTSEL = persel;	    // Passed DMA channel as peripheral interrupt source



92	DmaRegs.CH1.MODE.bit.PERINTE = perinte;       	// Peripheral interrupt enable



93	DmaRegs.CH1.MODE.bit.ONESHOT = oneshot;       	// Oneshot enable



94	DmaRegs.CH1.MODE.bit.CONTINUOUS = cont;    		// Continous enable



95	DmaRegs.CH1.MODE.bit.SYNCE = synce;         	// Peripheral sync enable/disable



96	DmaRegs.CH1.MODE.bit.SYNCSEL = syncsel;       	// Sync effects source or destination



97	DmaRegs.CH1.MODE.bit.OVRINTE = ovrinte;         // Enable/disable the overflow interrupt



98	DmaRegs.CH1.MODE.bit.DATASIZE = datasize;      	// 16-bit/32-bit data size transfers



99	DmaRegs.CH1.MODE.bit.CHINTMODE = chintmode;		// Generate interrupt to CPU at beginning/end of transfer



100	DmaRegs.CH1.MODE.bit.CHINTE = chinte;        	// Channel Interrupt to CPU enable



101



102	// Clear any spurious flags:



103	DmaRegs.CH1.CONTROL.bit.PERINTCLR = 1;  		// Clear any spurious interrupt flags



104	DmaRegs.CH1.CONTROL.bit.SYNCCLR = 1;    		// Clear any spurious sync flags



105	DmaRegs.CH1.CONTROL.bit.ERRCLR = 1; 	     	// Clear any spurious sync error flags



106



107	// Initialize PIE vector for CPU interrupt:



108	PieCtrlRegs.PIEIER7.bit.INTx1 = 1;              // Enable DMA CH1 interrupt in PIE



109



110	EDIS;



111}
