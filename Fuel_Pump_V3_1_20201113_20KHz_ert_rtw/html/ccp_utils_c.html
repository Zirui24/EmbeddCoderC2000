72void c_reset_all_DAQ_lists(void) {



73   int i;



74   /* loop through each DAQ list and reset them */

81/* resets the daq list specified by daq_number



82 * resets all elements in the list and sets the 



83 * daq_mode to DAQ_STOP */



84void c_reset_single_DAQ_list(uint8_T daq_number) {



85   /* test to see if DAQ lists are configured */



86   #ifdef CCP_DAQ_LIST_ENABLED



87      int i, j; /* loop counters */



88   



89      /* get pointer to single DAQ List */



90      C_DaqList* daq_ptr = &(c_dlist[daq_number]); 



91      /* set daq mode to STOP */



92      daq_ptr->daq_mode = DAQ_STOP;



93   



94      for (i=0; i < C_NUMBER_OF_ODTS_PER_DAQ; i++) {



95         /* get pointer to single ODT */



96         C_DaqODT* odt_ptr = &(daq_ptr->odt[i]);     



97         for (j=0; j<7; j++) {



98            /* zero the length of each element */



99            odt_ptr->element[j].length = 0;



100         }



101      }



102   #endif



103}

105/* Initialise a DAQ list */



106/* This function is responsible for setting bytes 3 and 4 of the 



107 * DTO for GET_DAQ_SIZE */



108void c_init_daq_list(uint8_T daq_number) {



109   /* check the daq_number is valid 



110    * valid daq number --->    0 <= daq_number < C_NUMBER_OF_DAQ_LISTS */



111   if ((int16_T)daq_number >= C_NUMBER_OF_DAQ_LISTS) {



112      /* return zero list size */



113      ccp.data[3] = 0;



114      /* set pid to 0 */



115      ccp.data[4] = 0;



116      return;



117   }



118 



119   /* this part is only applicable if DAQ lists are configured */



120   #ifdef CCP_DAQ_LIST_ENABLED



121      /* call function to reset the daq list */



122      c_reset_single_DAQ_list(daq_number);



123   



124      /* set DAQ list size */



125      ccp.data[3] = C_NUMBER_OF_ODTS_PER_DAQ;



126      /* calculate first PID of DAQ list */



127      ccp.data[4] = daq_number * C_NUMBER_OF_ODTS_PER_DAQ;



128   #endif



129}

160void c_start_stop_all(uint8_T start_stop) {



161   /* 



162    * if start_stop == DAQ_START_SYNCHRONIZED then start all DAQ lists that have daq_mode == DAQ_PREPARE 



163    * if start_stop == DAQ_STOP_ALL then stop all DAQ lists



164    */



165   /* only valid if DAQ lists are configured */



166   #ifdef CCP_DAQ_LIST_ENABLED



167      if (start_stop == DAQ_START_SYNCHRONIZED) {



168         /* loop through all DAQ lists looking for DAQ_MODE == DAQ_PREPARE */



169         int dlist_ptr;



170         for (dlist_ptr=0; dlist_ptr < C_NUMBER_OF_DAQ_LISTS; dlist_ptr++) {



171            if (c_dlist[dlist_ptr].daq_mode == DAQ_PREPARE) {



172               c_dlist[dlist_ptr].daq_mode = DAQ_START;



173            }



174         }



175      }



176      else {



177         /* stop all DAQ lists */



178         int dlist_ptr;



179         for (dlist_ptr=0; dlist_ptr < C_NUMBER_OF_DAQ_LISTS; dlist_ptr++) {



180            c_dlist[dlist_ptr].daq_mode = DAQ_STOP;



181         }



182      }



183   #endif



184}

186void c_fire_DAQs(uint8_T event_channel) {



187   /* only valid if DAQ lists are configured */



188   #ifdef CCP_DAQ_LIST_ENABLED



189      /* loop through all ODT's sending appropriate data */



190      int dlist_ptr;



191      for (dlist_ptr=0; dlist_ptr < C_NUMBER_OF_DAQ_LISTS; dlist_ptr++) {



192         /* check the mode for the current list */



193         if (c_dlist[dlist_ptr].daq_mode == DAQ_START) {



194            /* check the event channel for the current list */



195            if (c_dlist[dlist_ptr].event_channel == event_channel) {



196               int odt_ptr;



197               /* loop over appropriate ODT's */



198               for (odt_ptr=0; odt_ptr <= c_dlist[dlist_ptr].last_odt; odt_ptr++) {



199                  /* construct a message for the current odt by 



200                   * looping through the elements */



201                  int element_ptr;



202                  int msg_ptr = 0;



203                  uint8_T msg[8];



204                  /* set the pid for the message */



205                  msg[msg_ptr] = (dlist_ptr * C_NUMBER_OF_ODTS_PER_DAQ) + odt_ptr;



206                  msg_ptr++;



207                  for (element_ptr=0; element_ptr < 7; element_ptr++) {



208                     C_DaqElement element = c_dlist[dlist_ptr].odt[odt_ptr].element[element_ptr];



209                     



210                     /* element.address is a pointer to the desired element



211                        add element to the message depending the element le0ngth */



212                     switch (element.length) {



213                        case BYTE_ELEMENT: 



214                                     {



215                                         /* get a byte pointer to this address */



216                                         uint16_T * bytesrc = (uint16_T *) element.address;



217                                         msg[msg_ptr++] = bytesrc[0]; 



218                                         break;



219                                     }



220                        case WORD_ELEMENT:



221                                     {



222                                         /* 16 bit pointer to address */



223                                         uint16_T * bytesrc = (uint16_T *) element.address;



224                                         



225                                         /* read 2 bytes into msg */



226                                         msg[msg_ptr++] = bytesrc[0] & 0xff;



227                                         msg[msg_ptr++] = bytesrc[0] >> 8;



228                                         break;



229                                     }



230                        case LONG_ELEMENT:



231                                     { 



232                                         /* 32 bit pointer to address */



233                                         uint32_T * longsrc = (uint32_T *) element.address;



234                                         /* atomic element copy in case of interruption



235                                          * by a faster task that changes the data 



236                                          * NOTE: 32 bit copy instruction required 



237                                          * NOTE: This is only an issue if DAQ element 



238                                          * is assigned to incorrect DAQ list! */



239                                         uint32_T temp = *longsrc;



240



241                                         /* byte copy to deal with alignment issues in



242                                          * output message - we do not know where elements



243                                          * will be placed */



244                                         uint16_T * bytesrc = (uint16_T *) &temp;



245                                         



246                                         /* read 4 bytes into msg */



247                                         msg[msg_ptr++] = bytesrc[0] & 0xff;



248                                         msg[msg_ptr++] = bytesrc[0] >> 8;



249                                         msg[msg_ptr++] = bytesrc[1] & 0xff;



250                                         msg[msg_ptr++] = bytesrc[1] >> 8;



251                                         break;    



252                                     }



253                     } /* end switch */



254                  } /* end element for */     



255                  /* send the message for the ODT 



256                   * exported_ccp_daq_trigger is exported by a block in the model */



257                  exported_ccp_daq_trigger(&msg[0]);



258               } /* end odt for */



259            } /* end event_channel if */



260         } /* end daq_mode if */



261      } /* end dlist for */



262   #endif



263} /* end function */
