77                *((SPI_obj->spi_base_address) + SPIFFTX_REG_OFFSET) |= ((SPIFFTX_SPIFFENA_BITS) << SPIFFTX_SPIFFENA_SHIFT);



78                *((SPI_obj->spi_base_address) + SPIFFCT_REG_OFFSET) |= ((MW_SPI_FIFOTRANSMITDELAY & SPIFFCT_TXDLY_BITS) << SPIFFCT_TXDLY_SHIFT);

501/*Set the SPIMode: Clock polarity and clock phase*/



502void SPI_clockFormat(MW_SpiHandle SPI_obj, Uint16 clockPolarity, Uint16 clockPhase)



503{



504    if(SPI_obj->spi_base_address != 0)



505    {



506        *((SPI_obj->spi_base_address) + SPICCR_REG_OFFSET) &= (~(SPICCR_CLKPOLARITY_BITS << SPICCR_CLKPOLARITY_SHIFT));



507        *((SPI_obj->spi_base_address) + SPICTL_REG_OFFSET) &= (~(SPICTL_CLK_PHASE_BITS << SPICTL_CLK_PHASE_SHIFT));



508        *((SPI_obj->spi_base_address) + SPICCR_REG_OFFSET) |= ((clockPolarity & SPICCR_CLKPOLARITY_BITS) << SPICCR_CLKPOLARITY_SHIFT);



509        *((SPI_obj->spi_base_address) + SPICTL_REG_OFFSET) |= ((clockPhase & SPICTL_CLK_PHASE_BITS) << SPICTL_CLK_PHASE_SHIFT);



510    }



511}

513/*Read the SPI status registers for read and write status*/



514Uint16 SPI_status(MW_SpiHandle SPI_obj, SPI_statusVal type)



515{



516    Uint16 ret = 0;



517    if(SPI_obj->spi_base_address != 0)



518    {



519        switch(type)



520        {



521            case SPI_OVERRUN_FLAG:



522                ret = (*((SPI_obj->spi_base_address) + SPISTS_REG_OFFSET) >> SPISTS_OVERRUN_FLAG_SHIFT)& SPISTS_OVERRUN_FLAG_BITS;



523                break;



524            case SPI_INT_FLAG:



525                ret = (*((SPI_obj->spi_base_address) + SPISTS_REG_OFFSET) >> SPISTS_INT_FLAG_SHIFT) & SPISTS_INT_FLAG_BITS;



526                break;



527            case SPI_BUFFULL_FLAG:



528                ret = (*((SPI_obj->spi_base_address) + SPISTS_REG_OFFSET) >> SPISTS_BUFFULL_FLAG_SHIFT) & SPISTS_BUFFULL_FLAG_BITS;



529                break;



530            case SPI_TXFFST_FLAG:



531                ret = (*((SPI_obj->spi_base_address) + SPIFFTX_REG_OFFSET) >> SPIFFTX_TXFFST_SHIFT) & SPIFFTX_TXFFST_BITS ;



532                break;



533            case SPI_RXFFST_FLAG:



534                ret = (*((SPI_obj->spi_base_address) + SPIFFRX_REG_OFFSET) >> SPIFFRX_RXFFST_SHIFT) & SPIFFRX_RXFFST_BITS;



535                break;



536            case SPI_RXFFOVF_FLAG:



537                ret = (*((SPI_obj->spi_base_address) + SPIFFRX_REG_OFFSET) >> SPIFFRX_RXFFOVF_SHIFT) & SPIFFRX_RXFFOVF_BITS;



538                break;



539            default :



540                ret = 0;



541        }



542    }



543    return ret;



544}
