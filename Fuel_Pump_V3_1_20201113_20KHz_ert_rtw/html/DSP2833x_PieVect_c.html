180//---------------------------------------------------------------------------



181// InitPieVectTable: 



182//---------------------------------------------------------------------------



183// This function initializes the PIE vector table to a known state.



184// This function must be executed after boot time.



185//



186



187void InitPieVectTable(void)



188{



189	int16	i;



190	Uint32 *Source = (void *) &PieVectTableInit;



191	Uint32 *Dest = (void *) &PieVectTable;



192		



193	EALLOW;	



194	for(i=0; i < 128; i++)



195		*Dest++ = *Source++;	



196	EDIS;



197



198	// Enable the PIE Vector Table



199	PieCtrlRegs.PIECTRL.bit.ENPIE = 1;	



200			



201}
